\section{Introduction}\label{sec:intro}
% ======================================================================

Consider the class of nonnegative, integrable functions supported on
the interval $[-\tfrac14, \tfrac14]$:
\[
  \mathcal{F} \;=\; \Bigl\{
    f : \R \to \R_{\ge 0}
    \;\Big|\;
    \supp(f) \subseteq \bigl[-\tfrac14, \tfrac14\bigr],\;
    \int f > 0
  \Bigr\}.
\]

The \emph{autoconvolution} of $f$ is
\[
  (f * f)(t) \;=\; \int_{-\infty}^{\infty} f(x)\, f(t - x)\, dx.
\]
Since $\supp(f) \subseteq [-\tfrac14, \tfrac14]$, the
autoconvolution is supported on $[-\tfrac12, \tfrac12]$.

\begin{definition}
The \emph{autoconvolution constant} is
\[
  C_{1a} \;=\; \inf_{f \in \mathcal{F}} \;
  \frac{\displaystyle\max_{|t| \le 1/2}\, (f*f)(t)}%
       {\displaystyle\left(\int f\right)^{\!2}}.
\]
\end{definition}

By homogeneity, we may normalize $\int f = 1$, so
$C_{1a} = \inf_{f} \|f*f\|_{L^\infty}$.

The best known bounds are $1.2802 \leq C_{1a} \leq 1.5029$.  The
upper bound comes from exhibiting explicit functions; the lower bound
(Cloninger \& Steinerberger, 2017) requires showing that \emph{no}
function can achieve a peak below the threshold.  This is the subject
of our proof.

The proof strategy:
\begin{enumerate}
  \item Reduce from continuous functions to finite-dimensional bin averages.
  \item Discretize the simplex to a finite lattice, with error bounds.
  \item Exhaustively check every lattice point, using pruning and a
    multiscale cascade to make this tractable.
\end{enumerate}


% ======================================================================
\section{From Functions to Bin Averages}\label{sec:bin}
% ======================================================================

\subsection{The Partition}

Fix a positive integer $n$ (the ``half-bin count'').  Partition
$(-\tfrac14, \tfrac14)$ into $d = 2n$ equal bins:
\[
  I_j \;=\; \left(\frac{j}{4n},\; \frac{j+1}{4n}\right),
  \qquad j = -n, -n+1, \ldots, n-1.
\]
Each bin has width $\tfrac{1}{4n}$.  In the implementation, we
re-index as $0, 1, \ldots, d-1$ via $j \mapsto j + n$.

\subsection{Bin Averages}

For any nonnegative $f$ with $\supp(f) \subseteq (-\tfrac14,
\tfrac14)$ and $\int f = 1$, define the \emph{bin averages}:
\[
  a_j \;=\; 4n \int_{I_j} f(x)\, dx \;\ge\; 0,
  \qquad j = 0, \ldots, d-1.
\]

The factor $4n$ normalizes so that $a_j$ is the average value of $f$
on bin $I_j$.  The constraint $\int f = 1$ becomes
$\sum_{j=0}^{d-1} a_j = 4n$.

\begin{definition}
The \emph{continuous simplex} is
$A_n = \bigl\{ a \in \R_{\ge 0}^d \mid \sum a_j = 4n \bigr\}$.
\end{definition}

\subsection{The Lower Bound on $\|f*f\|_\infty$}

\begin{lemma}[Reduction to Bin Averages --- Lemma~1 of CS17]
\label{lem:bin-reduction}
For any $f \in \mathcal{F}$ with $\int f = 1$ and bin averages $a$:
\[
  \|f * f\|_{L^\infty}
  \;\ge\;
  \max_{\substack{2 \le \ell \le 2n \\[2pt] k}}
  \frac{1}{4n\ell}
  \sum_{\substack{0 \le i, j \le d-1 \\[1pt] k \le i+j \le k+\ell-2}}
  a_i\, a_j.
\]
\end{lemma}

\begin{proof}[Proof sketch]
The proof uses two ingredients:

\medskip
\noindent\textbf{1.\ Support of cross-convolutions.}
Let $f_i = f \cdot \mathbf{1}_{I_i}$.  Then $f = \sum_i f_i$ and
$(f*f)(t) = \sum_{i,j} (f_i * f_j)(t)$.  The convolution $f_i * f_j$
is supported on the Minkowski sum $I_i + I_j$, so the index $i + j$
determines which part of the convolution domain receives each
cross-term.

\medskip
\noindent\textbf{2.\ Pigeonhole.}
For any nonneg $h$ and interval $J$:
$\|h\|_{L^\infty} \ge \tfrac{1}{|J|} \int_J h$.

\medskip
\noindent\textbf{Combining.}  A window of $\ell$ consecutive
convolution-domain values covers $\ell - 1$ consecutive sum-indices
and spans a convolution-domain interval of length $\tfrac{\ell}{2n}$.
All cross-terms $f_i * f_j$ with $k \le i + j \le k + \ell - 2$ land
in this window, contributing total $L^1$ mass
$\sum \tfrac{a_i}{4n} \cdot \tfrac{a_j}{4n}$.  Applying pigeonhole:
\[
  \|f * f\|_{L^\infty}
  \;\ge\;
  \frac{1}{\ell / (2n)}
  \sum_{\substack{k \le i+j \le k+\ell-2}}
  \frac{a_i\, a_j}{(4n)^2}
  \;=\;
  \frac{1}{4n\ell}
  \sum_{\substack{k \le i+j \le k+\ell-2}}
  a_i\, a_j. \qedhere
\]
\end{proof}

\begin{remark}[Intuition]
The products $a_i a_j$ measure how much mass from bins $i$ and $j$
contributes to $f * f$.  The window argument says: if enough
product-mass concentrates in a short interval of the convolution
domain, the autoconvolution peak must be large there.
\end{remark}


% ======================================================================
\section{Discretization of the Simplex}\label{sec:disc}
% ======================================================================

Lemma~\ref{lem:bin-reduction} reduced the problem to minimizing over
the continuous simplex $A_n$.  We now discretize it to a finite lattice.

\subsection{The Lattice}

Fix a positive integer $m$ (the grid resolution).

\begin{definition}
The \emph{discrete lattice} is
\[
  B_{n,m} \;=\; \left\{
    b \in \left(\tfrac{1}{m}\N\right)^d
    \;\middle|\;
    \sum_{j=0}^{d-1} b_j = 4n
  \right\}.
\]
\end{definition}

Writing $c_j = m \cdot b_j$, the lattice consists of nonneg integer
vectors with $\sum c_j = S = 4nm$ --- the \emph{compositions} of $S$
into $d$ parts.  There are $\binom{S+d-1}{d-1}$ such compositions.

\begin{lemma}[$\tfrac{1}{m}$-net --- Lemma~2 of CS17]
\label{lem:lattice-approx}
For every $a \in A_n$, there exists $b \in B_{n,m}$ with
$\|a - b\|_\infty \le \tfrac{1}{m}$.
\end{lemma}

\begin{proof}[Proof sketch]
Round each $a_j$ to the nearest multiple of $1/m$ via a greedy
procedure maintaining a running deficit, forcing the sum constraint
with the last coordinate.  Each coordinate changes by at most $1/m$.
\end{proof}

\subsection{The Discretization Error Bound}

\begin{lemma}[Discretization Error --- Lemma~3 of CS17]
\label{lem:disc-error}
\[
  C_{1a} \;\ge\; b_{n,m} - \frac{2}{m} - \frac{1}{m^2},
\]
where $b_{n,m} = \min_{b \in B_{n,m}} \tv(b)$.
\end{lemma}

\begin{proof}
Write $a = b + \varepsilon$ with $\|\varepsilon\|_\infty \le 1/m$.
Then $f * f = g*g + g*\varepsilon + \varepsilon*g +
\varepsilon*\varepsilon$ where $g$ has bin averages $b$.  By Young's
inequality:
\begin{itemize}
  \item $\|g * \varepsilon\|_\infty \le \|g\|_1 \cdot
    \|\varepsilon\|_\infty \le 1/m$.
  \item $\|\varepsilon * \varepsilon\|_\infty \le
    \|\varepsilon\|_1 \cdot \|\varepsilon\|_\infty \le 1/m^2$
    (since $\|\varepsilon\|_1 \le d \cdot \tfrac{1}{4n} \cdot
    \tfrac{1}{m} = 1/m$).
\end{itemize}
So $\|f*f\|_\infty \ge \|g*g\|_\infty - 2/m - 1/m^2 \ge
\tv(b) - 2/m - 1/m^2$.
\end{proof}

\begin{remark}
The correction $2/m + 1/m^2$ is the price of discretization.  At
$m = 20$ it is $0.1025$; at $m = 50$ it is $0.0404$.
\end{remark}

\subsection{Dynamic (Per-Window) Correction}

The uniform correction can be tightened on a per-window basis.

\begin{lemma}[Dynamic Correction --- Equation~(1) of CS17]
\label{lem:dynamic-correction}
For a window at position $k$ with length $\ell$, let
$W_{\mathrm{int}} = \sum_{i \in \mathcal{B}(k,\ell)} c_i$ be the
total integer mass in contributing bins.  Then the effective threshold
for that window is:
\[
  T_{\ell,k} \;=\; c_{\mathrm{target}} + \frac{1 + 2\, W_{\mathrm{int}}}{m^2}
\]
rather than the uniform $T = c_{\mathrm{target}} + 2/m + 1/m^2$.
\end{lemma}

This is tighter because $W_{\mathrm{int}} \le m$ always, with
equality only when all bins contribute (wide windows centered in the
domain).  For narrow or edge-positioned windows, $W_{\mathrm{int}}$
is much smaller, making the threshold lower and pruning easier.


% ======================================================================
\section{The Windowed Test Value}\label{sec:testvalue}
% ======================================================================

\subsection{Definition}

\begin{definition}
For a mass vector $a = (a_0, \ldots, a_{d-1})$ with $a_i \ge 0$ and
$\sum a_i = 4n$, the \emph{autoconvolution} is the vector of length
$2d - 1$:
\[
  \conv[s] \;=\; \sum_{\substack{i + j = s \\[1pt] 0 \le i,j \le d-1}}
  a_i\, a_j,
  \qquad s = 0, 1, \ldots, 2(d-1).
\]
The \emph{test value} is
\[
  \tv(a) \;=\;
  \max_{\substack{2 \le \ell \le 2d \\[2pt] 0 \le k \le 2(d-1) - (\ell-2)}}
  \frac{1}{4n\ell}
  \sum_{s=k}^{k+\ell-2} \conv[s].
\]
\end{definition}

In words: compute the autoconvolution, slide a window of each width
$\ell$ across it, sum the $\ell - 1$ entries in each position,
normalize by $4n\ell$, and take the maximum.

\subsection{Efficient Computation via Prefix Sums}

Let $P[s] = \sum_{k=0}^{s} \conv[k]$ be the prefix sum.  Then the
window sum from $s_\text{lo}$ to $s_\text{hi}$ is
$P[s_\text{hi}] - P[s_\text{lo} - 1]$ (with $P[-1] = 0$).

Computing the autoconvolution takes $O(d^2)$, prefix sums $O(d)$, and
all windows $O(d^2)$.  Total: $O(d^2)$ per lattice point.

\subsection{Reversal Symmetry}

This is the mathematical basis for the canonical optimization that
halves the search space.

\begin{theorem}[Autoconvolution Reversal Symmetry]
\label{thm:reversal}
For any mass vector $a$, $\tv(a) = \tv(\rev(a))$, where
$\rev(a) = (a_{d-1}, \ldots, a_0)$.
\end{theorem}

\begin{proof}
\textbf{Step 1: Autoconvolution reversal.}
Let $a' = \rev(a)$, so $a'_i = a_{d-1-i}$.  Then:
\[
  \conv'[s]
  \;=\; \sum_{i+j=s} a'_i a'_j
  \;=\; \sum_{i+j=s} a_{d-1-i}\, a_{d-1-j}.
\]
Substituting $i' = d-1-i$, $j' = d-1-j$ gives $i'+j' = 2(d-1) - s$,
so $\conv'[s] = \conv[2(d-1) - s]$.  The autoconvolution of the
reversal is the reversal of the autoconvolution.

\medskip
\noindent\textbf{Step 2: Window-sum bijection.}
A window sum of width $w = \ell - 1$ at position $s_0$ transforms to:
\[
  \sum_{s=s_0}^{s_0+w-1} \conv'[s]
  = \sum_{s'=2(d-1)-s_0-w+1}^{2(d-1)-s_0} \conv[s'].
\]
This is a window of the \emph{same width} $w$ at a reflected
position.  The mapping is a bijection on windows of each width.

\medskip
\noindent\textbf{Step 3: Normalization preserves the maximum.}
Since $\tfrac{1}{4n\ell}$ depends only on $\ell$ (not the position),
and the multiset of window sums at each width is identical for
$\conv$ and $\conv'$, the maximum is the same:
$\tv(a) = \tv(a')$.
\end{proof}

\begin{corollary}[Canonical Symmetry]
It suffices to check only \emph{canonical} compositions satisfying
$b \le \rev(b)$ lexicographically.  This halves the search space
(up to palindromes).
\end{corollary}

\section{Pruning Strategies}\label{sec:pruning}
% ======================================================================

The lattice $B_{n,m}$ has $\binom{S + d - 1}{d - 1}$ points.
Pruning eliminates configurations cheaply, without the full test value.

\subsection{Asymmetry Pruning}

\begin{theorem}[Asymmetry Bound]
\label{thm:asymmetry}
Let $f \ge 0$ with $\supp(f) \subseteq (-\tfrac14, \tfrac14)$,
$\int f = 1$, and left-half mass $L = \int_{-1/4}^{0} f$.  Then
$\|f * f\|_{L^\infty} \ge 2L^2$.
\end{theorem}

\begin{proof}
The left-left contribution of $f*f$ on $(-\tfrac12, 0)$ has total
$L^1$ mass at least $L^2$ (by restricting both integration variables
to $(-\tfrac14, 0)$ in the convolution integral).  Since
$|(-\tfrac12, 0)| = \tfrac12$, the pigeonhole bound gives
$\|f*f\|_\infty \ge L^2 / (1/2) = 2L^2$.  By symmetry, the same
holds for the right-half mass $R = 1 - L$.
\end{proof}

\begin{corollary}[Asymmetry Threshold]
If $L \ge \sqrt{c_{\mathrm{target}}/2}$, the configuration is
eliminated.  In the discrete setting, a margin of $1/(4m)$ is added
to account for rounding:
$p^*_{\mathrm{safe}} = \sqrt{c_{\mathrm{target}}/2} + 1/(4m)$.
\end{corollary}

\begin{remark}
The asymmetry bound is a direct $L^\infty$ bound --- it does
\emph{not} go through discretization, so we compare against
$c_{\mathrm{target}}$ itself (not $c_{\mathrm{target}} + 2/m + 1/m^2$).
\end{remark}

\subsection{Contiguous Block-Sum Bound}

\begin{theorem}[Block-Sum Bound]
\label{thm:block-sum}
For a contiguous block of $p$ bins $\{q, \ldots, q+p-1\}$ with sum
$B = \sum_{i=q}^{q+p-1} c_i$:
\[
  \tv(a) \;\ge\; \frac{B^2}{4n \cdot 2p \cdot m^2}.
\]
\end{theorem}

\begin{proof}
The window $\ell = 2p$ at $k = 2q$ captures all intra-block pairs
$(i,j)$ with $q \le i,j \le q+p-1$.  Since $a_i \ge 0$, cross-block
terms only add, so the window sum $\ge (\sum_{\text{block}} a_i)^2
= B^2/m^2$.  Normalizing by $4n \cdot 2p$ gives the result.
\end{proof}

\begin{remark}[Special cases]
$p = 1$ gives the single-element bound $c_i^2 / (8nm^2)$.
$p = n$ gives the half-sum bound (related to asymmetry).
Intermediate $p$ catches mass concentrated in middle blocks
that evade both checks.
\end{remark}

\subsection{Two-Max Cross-Term Bound}

\begin{theorem}[Two-Max Bound]
\label{thm:two-max}
Let $c_{p_1}, c_{p_2}$ be the two largest values with $p_1 \ne p_2$.
Then:
\[
  \tv(a) \;\ge\; \frac{2\, c_{p_1}\, c_{p_2}}{8n\, m^2}.
\]
\end{theorem}

\begin{proof}
At index $s = p_1 + p_2$: $\conv[s] \ge a_{p_1} a_{p_2} +
a_{p_2} a_{p_1} = 2 c_{p_1} c_{p_2} / m^2$.  This single entry is
an $\ell = 2$ window, giving $\tv \ge 2 c_{p_1} c_{p_2} / (8nm^2)$.
\end{proof}

\subsection{Canonical Counting}

\begin{proposition}[Canonical Count]
\label{prop:canon-count}
The number of canonical compositions ($b \le \rev(b)$ lex) is
$\tfrac{N + P}{2}$, where $N = \binom{S+d-1}{d-1}$ and $P$ is the
palindrome count.
\end{proposition}

\begin{proof}
Partition compositions into palindromes ($P$), strictly canonical
non-palindromes ($C$), and strictly non-canonical ($C$).  The last
two classes biject via reversal, so $N = P + 2C$ and the canonical
count is $P + C = (N+P)/2$.
\end{proof}

\begin{remark}[Loop-bound tightening for $d = 4$]
For $(c_0, c_1, c_2, c_3)$ with $c_3 = S - c_0 - c_1 - c_2$,
canonicality requires $c_0 \le c_3$, yielding tight loop bounds:
$c_0 \le \lfloor S/2 \rfloor$,
$c_1 \le S - 2c_0$,
$c_2 \le S - 2c_0 - c_1$,
plus the palindrome filter $c_0 = c_3 \Rightarrow c_1 \le c_2$.
Analogous (but more complex) bounds hold for $d = 6$.
\end{remark}

\section{The Multiscale Cascade}\label{sec:cascade}
% ======================================================================

Rather than enumerate all compositions at fine resolution directly,
the cascade starts coarse and refines only survivors.

\subsection{Dyadic Splitting}

A parent $b = (b_0, \ldots, b_{d-1})$ at resolution $d$ produces
children $c = (c_0, \ldots, c_{2d-1})$ at resolution $2d$, where
$c_{2i} + c_{2i+1} = b_i$ for each $i$, with both parts nonneg
integers.

\subsection{Energy Cap}

A child bin $c_j$ exceeding
$x_{\mathrm{cap}} = \lfloor m \sqrt{T / d_{\mathrm{child}}} \rfloor$
would be immediately pruned by the $\ell = 2$ diagonal window, so we
never generate it.

\begin{proposition}[Energy Cap Soundness]
If $c_j > x_{\mathrm{cap}}$, then the $\ell=2$ test value at
position $2j$ exceeds $T$.
\end{proposition}

\begin{proof}
With $a_j = c_j \cdot 4n_{\mathrm{child}} / m$ and
$d_{\mathrm{child}} = 2n_{\mathrm{child}}$:
\[
  \frac{a_j^2}{4 n_{\mathrm{child}} \cdot 2}
  = \frac{c_j^2 \cdot 2 n_{\mathrm{child}}}{m^2}
  > \frac{m^2 \cdot T/(2n_{\mathrm{child}}) \cdot 2n_{\mathrm{child}}}{m^2}
  = T. \qedhere
\]
\end{proof}

\subsection{Cascade Soundness}

\begin{theorem}[Cascade Soundness]
\label{thm:cascade-sound}
If the cascade terminates with zero survivors at level $L_k$, then
$C_{1a} \ge c_{\mathrm{target}}$.
\end{theorem}

\begin{proof}
By induction, every composition in $B_{n_k, m}$ is either directly
pruned or an ancestor of pruned descendants.

At $L_0$, all canonical compositions are enumerated (reversal
symmetry covers non-canonicals).  At each subsequent level, \emph{all}
children of each surviving parent are generated and tested.  If every
child is pruned, the parent is indirectly eliminated.

When zero survivors remain: every $b \in B_{n_k, m}$ has
$\tv(b) \ge T$.  For any $f$ with bin averages $a$, there exists
$b$ with $\|a - b\|_\infty \le 1/m$ (Lemma~\ref{lem:lattice-approx}).
Then:
\[
  \|f*f\|_\infty \ge \tv(a) \ge \tv(b) - \frac{2}{m} - \frac{1}{m^2}
  \ge T - \frac{2}{m} - \frac{1}{m^2} = c_{\mathrm{target}}. \qedhere
\]
\end{proof}

\subsection{Canonical Handling Across Levels}

At $L_0$, only canonical compositions are generated.  At $L_1{+}$:

\begin{itemize}
  \item Generate \emph{all} children (no canonical filter during
    generation), because children of canonical $P$ need not be
    canonical.
  \item Test each child.  Since $\tv(C) = \tv(\rev(C))$, testing
    $C$ covers $\rev(C)$ automatically.
  \item Canonicalize survivors: $C \mapsto \min(C, \rev(C))$.
  \item Deduplicate (sort + scan).
\end{itemize}

This is complete because if $C'$ is any child-level composition, then
either its parent or the parent's reversal is in the parent list, so
$C'$ or $\rev(C')$ is tested.

\subsection{Fused Generate-and-Prune Kernel}

At deep cascade levels, each parent may have millions of children.
The fused kernel generates and tests each child inline via a
Cartesian-product odometer, never materializing the full array:

\begin{itemize}
  \item \textbf{Incremental autoconvolution:} when only the last
    cursor position changes ($\sim$67\% of steps), the raw
    autoconvolution is updated in $O(d)$ instead of $O(d^2)$.
  \item \textbf{Quick-check:} re-tries the previous killing window
    before a full scan.
  \item \textbf{Subtree pruning:} on deep carries, checks the
    partial autoconvolution of fixed bins against worst-case unfixed
    mass; if it exceeds the threshold, the entire subtree is skipped.
\end{itemize}
